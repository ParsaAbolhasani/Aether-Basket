pragma solidity ^0.8.25;

import "forge-std/Test.sol";
import "../src/Basket.sol";

contract BasketTest is Test {
    Basket public basket;

    function setUp() public {
        // آدرس موقت برای اوراکل (در مرحله بعد جایگزین می‌شود)
        basket = new Basket(address(1), address(this));
    }

    function test_AddAsset() public {
        basket.addAsset(address(0), 1000);
        assertEq(basket.getCurrentAllocations().length, 1);
    }
}
