pragma solidity ^0.8.25;

import "forge-std/Test.sol";
import "../../src/Basket.sol";
import "../../src/oracle/ChainlinkOracle.sol";

contract BasketFuzzTest is Test {
    Basket public basket;
    ChainlinkOracle public oracle;

    function setUp() public {
        oracle = new ChainlinkOracle(address(this));
        basket = new Basket(address(oracle), address(this));
    }

    function testFuzz_AddAsset(uint256 targetPercent) public {
        targetPercent = bound(targetPercent, 1, 10000);
        basket.addAsset(address(0x123), targetPercent);
        
        (address token, uint256 percent) = basket.assets(0);
        assertEq(token, address(0x123));
        assertEq(percent, targetPercent);
    }

    function testFuzz_MultipleAssets(uint8 count) public {
        count = uint8(bound(count, 1, 20));
        uint256 totalPercent = 0;
        
        for (uint8 i = 0; i < count; i++) {
            uint256 percent = uint256(bound(rand(), 1, 10000));
            basket.addAsset(address(uint160(i + 1)), percent);
            totalPercent += percent;
        }
        
        assertEq(basket.getAssetCount(), count);
        // مجموع درصدها می‌تواند بیشتر از 10000 باشد، اما در Rebalance چک می‌شود
    }
}
